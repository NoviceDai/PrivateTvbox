```yaml
# 自定义域名规则 (geosite 分类)

# GFWList 封锁站点
DOMAIN-SUFFIX,jellyfin.org
DOMAIN-SUFFIX,netflav.com

DOMAIN-SUFFIX,xvideos.com
DOMAIN-SUFFIX,pornhub.com
DOMAIN-SUFFIX,xhamster.com

# .buzz 混淆域名
DOMAIN-SUFFIX,*.buzz

# GitHub (配置文件来源)
DOMAIN-SUFFIX,github.com
DOMAIN-SUFFIX,githubusercontent.com

# 境外成人站点
DOMAIN-KEYWORD,porn
DOMAIN-KEYWORD,jav
DOMAIN-KEYWORD,av
```
